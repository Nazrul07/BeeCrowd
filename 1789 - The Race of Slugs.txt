#include<stdio.h>
int main()
{
    int n,i,a,Max;
    while(scanf("%d",&n)!=EOF)
    {
        Max=0;
        for(i=1 ; i<=n ; i++)
        {
            scanf("%d",&a);
            if(Max<a)
                Max=a;
        }
        if(Max<10)
            printf("1\n");
        else if(Max>=10 && Max<20)
            printf("2\n");
        else
            printf("3\n");
    }
    return 0;
}
