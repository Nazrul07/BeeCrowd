#include<stdio.h>
int main()
{
    float a,b,c,d,e,f;
    int count=0;
    scanf("%f%f%f%f%f%f",&a,&b,&c,&d,&e,&f);
    if(a>=0)
        count++;
    if(b>=0)
        count++;
    if(c>=0)
        count++;
    if(d>=0)
        count++;
    if(e>=0)
        count++;
    if(f>=0)
        count++;
    printf("%d valores positivos\n",count);
    return 0;
}
