#include<stdio.h>
int main()
{
    int n,f1,f2,i,sum;
    scanf("%d",&n);
    f1=0;
    f2=1;
    for(i=1 ; i<=n ; i++)
    {
        if(i==n)
            printf("%d",f1);
        else
            printf("%d ",f1);
        sum=f1+f2;
        f1=f2;
        f2=sum;
    }
    printf("\n");
    return 0;
}
