#include<stdio.h>
int main()
{
    int code;
    float amu;
    scanf("%d%f",&code,&amu);
    switch(code)
    {
    case 1:
        printf("Total: R$ %.2f\n",4.00*amu);
        break;
    case 2:
        printf("Total: R$ %.2f\n",4.50*amu);
        break;
    case 3:
        printf("Total: R$ %.2f\n",5.00*amu);
        break;
    case 4:
        printf("Total: R$ %.2f\n",2.00*amu);
        break;
    case 5:
        printf("Total: R$ %.2f\n",1.50*amu);
        break;
    }
}
