#include<stdio.h>
int main()
{
    int start,end,total;
    scanf("%d%d",&start,&end);
    if(start<end)
        printf("O JOGO DUROU %d HORA(S)\n",end-start);
    else if(start==end)
        printf("O JOGO DUROU 24 HORA(S)\n");
    else
    {
        total=(24-start)+end;
        printf("O JOGO DUROU %d HORA(S)\n",total);
    }
    return 0;
}
