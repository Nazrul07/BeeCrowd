#include<bits/stdc++.h>
using namespace std;
int main()
{
    int a,b;
    char c;
    while(cin>>a>>c>>b)
    {
        if(a<7)
            cout<<"Atraso maximo: 0"<<endl;
        else
        {
            int count=(a-7)*60+b;
            cout<<"Atraso maximo: "<<count<<endl;
        }
    }
    return 0;
}
