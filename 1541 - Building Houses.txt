#include<stdio.h>
int main()
{
    int i,a,b,c,total,val,ans;
    while(1)
    {
        scanf("%d",&a);
        if(a==0) break;
        scanf("%d%d",&b,&c);
        total=(a*b*100)/c;
        for(i=1 ; ; i++)
        {
            if(i*i<=total)
             ans=i;
            else break;
        }
        printf("%d\n",ans);
    }
}
